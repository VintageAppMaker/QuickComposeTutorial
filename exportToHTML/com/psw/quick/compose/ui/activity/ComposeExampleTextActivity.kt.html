57       @Composable
58       fun ExampleMain(
59       ) {
60           val weightLst = listOf(
61               Pair("ExtraBold", FontWeight.ExtraBold),
62               Pair("Bold", FontWeight.Bold),
63               Pair("Medium", FontWeight.Medium),
64               Pair("Normal", FontWeight.Normal),
65               Pair("Light", FontWeight.Light),
66               Pair("Thin", FontWeight.Thin)
67           )
68   
69           val ellipsisLst = listOf(
70               Pair("Ellipsis - Text", TextOverflow.Ellipsis),
71               Pair("Clip - Text", TextOverflow.Clip),
72               Pair("Visible - Text", TextOverflow.Visible)
73           )
74   
75           val scrState = rememberScrollState()
76   
77           // Text의 크기는 SP가 기본.
78           Column (modifier = Modifier
79               .padding(20.dp)
80               .background(Color.White)
81               .verticalScroll(scrState)){
82   
83   
84               // FontWeight 예제.
85               Text("FontWeight 리스트", color = Color.Black, fontSize = 18.sp)
86               Divider(color = Color.Gray, thickness = 0.5.dp)
87               Spacer(Modifier.height(10.dp))
88               weightLst.forEach { item ->
89                   Text("${item.first}", fontWeight = item.second, color = Color.Gray)
90               }
91   
92               // ellipsis 예제. - 범위를 넘어가는 문자열
93               Text("ellipsis 리스트", color = Color.Black, fontSize = 18.sp)
94               Divider(color = Color.Gray, thickness = 0.5.dp)
95               Spacer(Modifier.height(10.dp))
96               ellipsisLst.forEach { item ->
97                   Box(modifier = Modifier.width(55.dp)){
98                       Text("${item.first}", maxLines = 1, overflow = item.second, color = Color.Gray)
99                   }
100              }
101  
102              // TextStyle 예제. TextStyle 사용하기
103              Text("TextStyle", color = Color.Black, fontSize = 18.sp)
104              Divider(color = Color.Gray, thickness = 0.5.dp)
105              Spacer(Modifier.height(10.dp))
106  
107              Text("style= TextStyle(...)", style = TextStyle(
108                  shadow = Shadow(
109                      color = Color.White,
110                      offset = Offset(5f, 5f),
111                      blurRadius = 5f
112                  ),
113                  color = Color.LightGray,
114                  background = Color.Black,
115                  fontSize = 30.sp,
116                  fontWeight = FontWeight.Thin,
117                  fontStyle = FontStyle.Italic
118  
119              ))
120  
121          }
