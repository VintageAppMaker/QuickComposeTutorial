59       @OptIn(ExperimentalFoundationApi::class)
60       @Composable
61       fun ExampleMain() {
62           // LazyColumn에서 item을 사용하여
63           // NestedScroll처럼 사용
64           LazyColumn(modifier = Modifier.fillMaxSize()) {
65   
66               // Header
67               stickyHeader(key = "List") {
68                   Text("List Test",
69                       style = TextStyle(fontSize = 30.sp, color = Color.White),
70                       modifier = Modifier
71                           .fillMaxWidth()
72                           .background( Color.Black.copy(alpha = 0.7f)),
73                       textAlign = TextAlign.Center
74                   )
75               }
76   
77               // ex1) list를 items로 처리함
78               items(items = bigData, itemContent = { item ->
79                   Text("$item", modifier = Modifier.padding(start=20.dp),
80                       style = TextStyle(fontSize = 15.sp, color = Color.Black))
81               })
82   
83               // Header
84               stickyHeader(key = "Row") {
85                   Text("Row Test",
86                       style = TextStyle(fontSize = 30.sp, color = Color.White),
87                       modifier = Modifier
88                           .fillMaxWidth()
89                           .background( Color.Red.copy(alpha = 0.7f)),
90                       textAlign = TextAlign.Center
91                   )
92               }
93   
94               // ex2) item로 둘러쌓은 후, LazyRow를 사용하여 list 처리
95               item {
96                   LazyRow{
97                       // item으로 섹션처리(LazyRow 예제)
98                       items(items = bigData2, itemContent = { item ->
99                           Box( modifier = Modifier
100                              .height(120.dp)
101                              .width(120.dp)
102                              .padding(5.dp)
103                              .background(Color.Black)){
104                              Text("$item",
105                                  modifier=Modifier.align(Alignment.Center),
106                                  color = Color.White, textAlign = TextAlign.Center)
107                          }
108                      })
109                  }
110              }
111  
112              // Header
113              stickyHeader(key = "Grid") {
114                  Text("Grid Test",
115                      style = TextStyle(fontSize = 30.sp, color = Color.White),
116                      modifier = Modifier
117                          .fillMaxWidth()
118                          .background( Color.Blue.copy(alpha = 0.7f)),
119                      textAlign = TextAlign.Center
120                  )
121              }
122  
123              // 제공되는 Grid가 아직까지 실험상태에 있다. 그래서
124              // Google의 Accompanist FlowLayout library를 사용하여 구현해야한다.
125              item{
126                  val gridSize: Dp = (LocalConfiguration.current.screenWidthDp.dp / 2)
127                  FlowRow(mainAxisSize = SizeMode.Expand,
128                      mainAxisAlignment = FlowMainAxisAlignment.SpaceBetween) {
129                      bigData3.forEach {
130                          item ->
131                          Card(modifier = Modifier
132                              .size(gridSize)
133                              .padding(5.dp), backgroundColor = Color.LightGray){
134                              Box(modifier = Modifier
135                                  .fillMaxSize()
136                                  .background(Color.Red)) {
137                                  Text("${item}",
138                                      style = TextStyle(color = Color.White),
139                                      modifier =Modifier.align(Alignment.Center),
140                                      textAlign = TextAlign.Center)
141                              }
142                          }
143                      }
144                  }
145              }
146          }
147      }
