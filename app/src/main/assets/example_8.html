45       // remember{}를 사용할 경우, Composable은 상태관리를 하는 Stateful이 된다.
46       // Stateful은 성능저하를 가져온다(해당함수가 전체갱신됨)
47       // 최소한으로 만들기 위해서 hoisting을 사용한다(부분갱신).
48       // hoisting은 Composable 함수에 remember를 가지지않고
49       // 외부에서 파라메터를 받아처리하는 방법이다.
50       @Composable
51       fun ExampleMain(
52       ) {
53           val scrState = rememberScrollState()
54           var count = remember{ mutableStateOf(0)}
55   
56           Column (modifier = Modifier
57               .padding(20.dp)
58               .background(Color.White)
59               .verticalScroll(scrState)){
60   
61               // remember 사용예제
62               Text("remeber 사용", color = Color.Black, fontSize = 18.sp)
63               Divider(color = Color.Gray, thickness = 0.5.dp)
64               Spacer(Modifier.height(10.dp))
65   
66               rememberStateExample()
67   
68               // hoisting 사용예제
69               Text("hoisting 사용", color = Color.Black, fontSize = 18.sp)
70               Divider(color = Color.Gray, thickness = 0.5.dp)
71               Spacer(Modifier.height(10.dp))
72   
73               hoistingExample(count.value, {count.value++})
74           }
75       }
76   
77       @Composable
78       fun rememberStateExample(){
79           var count = remember{ mutableStateOf(0)}
80           Row(Modifier.fillMaxWidth()){
81               Text("${count.value}",
82                   color = Color.Blue,
83                   modifier = Modifier.weight(1.0f)
84                       .fillMaxHeight().align(Alignment.CenterVertically),
85                   textAlign = TextAlign.Center)
86   
87               Button(onClick = {count.value++}, modifier = Modifier.weight(1.0f).padding(end= 5.dp)){
88                   Text("click +")
89               }
90           }
91       }
92   
93       @Composable
94       fun hoistingExample(count : Int, fnOnChange : ()->Unit){
95           Row(Modifier.fillMaxWidth()){
96               Text("${count}",
97                   color = Color.Blue,
98                   modifier = Modifier.weight(1.0f)
99                       .fillMaxHeight().align(Alignment.CenterVertically),
100                  textAlign = TextAlign.Center)
101  
102              Button(onClick = {fnOnChange()}, modifier = Modifier.weight(1.0f).padding(end= 5.dp)){
103                  Text("click +")
104              }
105          }
106      }
