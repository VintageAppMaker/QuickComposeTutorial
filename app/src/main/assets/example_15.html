26   // 프로젝트를 신규로 만들시
27   // MVVM은 기본구조이기에
28   // ViewModel에서 비지니스 로직을 코루틴으로 구현하는 것은
29   // 일반화 되어있다.
30   class ExampleViewModel : ViewModel(){
31       var title = mutableStateOf("")
32       var count = mutableStateOf(0)
33   
34       // StateFlow를 사용한다.
35       private val _State = MutableStateFlow<ActionState>(ActionState.Nodata)
36       val actionState: StateFlow<ActionState> = _State
37   
38       // 통신함수를 가상으로 구현
39       fun getData(){
40           CoroutineScope(Dispatchers.IO).launch {
41               _State.value = ActionState.Nodata
42   
43               delay(2000)
44               _State.value = ActionState.Loading
45   
46               delay(2000)
47               _State.value = ActionState.Loaded(
48                   (0..10).map { "data => $it" }
49                       .toList()
50               )
51           }
52       }
53   
54       // 액션 상태정의
55       sealed class ActionState {
56           object Nodata : ActionState()
57           object Loading : ActionState()
58           class Loaded(val data: List<String>) : ActionState()
59           class Error(val message: String) : ActionState()
60       }
61   }

87       @Composable
88       fun ExampleMain(
89           exampleViewModel: ExampleViewModel = ExampleViewModel() // 최상위 Composable 함수에만 사용
90       ) {
91           val scrStateCol  = rememberScrollState()
92   
93           Column (modifier = Modifier
94               .padding(20.dp)
95               .background(Color.White)
96               .verticalScroll(scrStateCol),
97               horizontalAlignment = Alignment.CenterHorizontally
98           ){
99   
100              //  viewModel의 StateFlow값을 대기
101              when( val rst = exampleViewModel.actionState.collectAsState().value){
102                  is ExampleViewModel.ActionState.Nodata ->{
103                      Text("데이터가 없습니다",
104                          style = TextStyle(fontSize = 30.sp, color = Color.Black),
105                          textAlign = TextAlign.Center)
106                  }
107  
108                  is ExampleViewModel.ActionState.Loading ->{
109                      Text("로딩중입니다. ",
110                          style = TextStyle(fontSize = 30.sp, color = Color.Black),
111                          textAlign = TextAlign.Center)
112  
113                      LinearProgressIndicator()
114                  }
115  
116                  is ExampleViewModel.ActionState.Loaded ->{
117                      Text("결과",
118                          style = TextStyle(color = Color.Black, fontSize = 30.sp ),
119                          textAlign = TextAlign.Center)
120                      Divider(thickness = 3.dp)
121  
122                      // ActionState.Loaded로 넘겨진 데이터는
123                      // data로 액세스 가능
124                      rst.data.forEach {
125                          Text("$it", style = TextStyle(color = Color.Black))
126                      }
127                      Divider(thickness = 3.dp)
128                  }
129              }
130  
131              Spacer(modifier = Modifier.height(20.dp))
132  
133              // viewModel의 인스턴스를 
134              // 다른 컴포저블 함수에 넘기면 안된다.(공식문서에서 강하게 언급)
135              // 메모리 누수 발생가능성이 높아지기에 
136              // viewModel의 값을 파라메터로 넘기는 호이스팅을 사용한다. 
137              buildCounter(exampleViewModel.count.value, {exampleViewModel.count.value++})
138              buildRestartButton(exampleViewModel::getData)
139          }
140  
141          // 최초1회 실행
142          LaunchedEffect(key1 = Unit){
143              exampleViewModel.getData()
144          }
145      }
146  
147      @Composable
148      private fun buildCounter(count: Int, fnInc : ()->Unit) {
149          Text("count : ${count}",
150              style = TextStyle(fontSize = 30.sp, color = Color.Black),
151              textAlign = TextAlign.Center)
152          Spacer(modifier = Modifier.height(20.dp))
153          Button(onClick = { fnInc() }, colors = ButtonDefaults.buttonColors(backgroundColor = Color.White)) {
154              Text("inc ++",
155                  style = TextStyle(fontSize = 15.sp, color = Color.Black),
156                  textAlign = TextAlign.Center)
157          }
158          Spacer(modifier = Modifier.height(20.dp))
159      }
160  
161      @Composable
162      private fun buildRestartButton(fnRestart : ()->Unit) {
163          Divider(thickness = 3.dp)
164          Spacer(modifier = Modifier.height(20.dp))
165          Button(onClick = { fnRestart() }, colors = ButtonDefaults.buttonColors(backgroundColor = Color.White)) {
166              Text("getData()",
167                  style = TextStyle(fontSize = 15.sp, color = Color.Black),
168                  textAlign = TextAlign.Center)
169          }
170      }
