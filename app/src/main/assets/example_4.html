68       @Composable
69       private fun buildDynamicLayout() {
70           BoxWithConstraints {
71               if (maxHeight < 150.dp) {
72                   buildRowLayout()
73               } else {
74                   buildConstraintLayout()
75               }
76           }
77       }
78   
79       @Composable
80       fun buildConstraintLayout() {
81           fun setConstraints(): ConstraintSet {
82               return ConstraintSet {
83                   val box1 = createRefFor("box1")
84                   val box2 = createRefFor("box2")
85   
86                   constrain(box1) {
87                       top.linkTo(parent.top, margin = 30.dp)
88                       start.linkTo(parent.start)
89                       end.linkTo(parent.end)
90   
91                       // 늘린크기만큼 width
92                       width = Dimension.fillToConstraints
93                   }
94   
95                   constrain(box2) {
96                       top.linkTo(box1.bottom, margin = 10.dp)
97                   }
98   
99               }
100          }
101  
102          ConstraintLayout(setConstraints()) {
103              Box(
104                  Modifier
105                      .background(Color.Red)
106                      .fillMaxWidth()
107                      .height(100.dp)
108                      .layoutId("box1"))
109  
110              Box(
111                  Modifier
112                      .background(Color.Green)
113                      .width(100.dp)
114                      .height(100.dp)
115                      .layoutId("box2"))
116          }
117      }
